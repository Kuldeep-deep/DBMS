GROUP BY Clause with Multiple Columns

    mysql> SELECT ADDRESS, AGE, SUM(SALARY) AS TOTAL_SALARY FROM CUSTOMERS GROUP BY ADDRESS, AGE;

GROUP BY with ORDER BY Clause

    mysql> SELECT AGE, MIN(SALARY) AS MIN_SALARY FROM CUSTOMERS GROUP BY AGE ORDER BY MIN_SALARY DESC;
+-----+------------+
| AGE | MIN_SALARY |
+-----+------------+
|  35 |   15000.00 |
|  29 |   13000.00 |
|  27 |    7500.00 |
|  28 |    5000.00 |
|  30 |    4500.00 |
+-----+------------+
5 rows in set (0.00 sec)

    
GROUP BY with HAVING Clause

    mysql> SELECT ADDRESS, AGE, MIN(SALARY) AS MIN_SUM FROM CUSTOMERS GROUP BY ADDRESS, AGE HAVING AGE>24;
