DECLARE  
   a integer := 30;  
   b integer := 40;  
   c integer;  
   f real;  
BEGIN  
   c := a + b;  
   dbms_output.put_line('Value of c: ' || c);  
   f := 100.0/3.0;  
   dbms_output.put_line('Value of f: ' || f);  
END;  
    
OUTPUT

    Value of c: 70
Value of f: 33.333333333333333333

PL/SQL procedure successfully completed.
